001    package realtimeweb.earthquakeservice.domain;
002    
003    /**
004     * Note that the threshold indicates the MINIMUM level that earthquakes are
005     * reported; so using Threshold.MINOR will yield all earthquakes that are 2.5+!
006     * 
007     * @author acbart
008     * 
009     */
010    public enum Threshold {
011            SIGNIFICANT, ALL, MICRO, MINOR, LIGHT;
012    
013            public String toString() {
014                    switch (this) {
015                    case SIGNIFICANT:
016                            return "significant";
017                    case ALL:
018                            return "all";
019                    case MICRO:
020                            return "1.0";
021                    case MINOR:
022                            return "2.5";
023                    case LIGHT:
024                            return "4.5";
025                    default:
026                            return "all";
027                    }
028            }
029    }
