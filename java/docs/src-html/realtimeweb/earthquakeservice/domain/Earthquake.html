373            /**
374             * Internal constructor to create a Earthquake from a Json representation.
375             * 
376             * @param json
377             *            The raw json data that will be parsed.
378             * @param gson
379             *            The Gson parser. See <a href=
380             *            'https://code.google.com/p/google-gson/'>https://code.google.com/p/google-gson/</a>
381             *            for more information.
382             * @return
383             */
384            public Earthquake(JsonObject json, Gson gson) {
385                    this.location = new Coordinate(json.get("geometry").getAsJsonObject()
386                                    .get("coordinates").getAsJsonArray(), gson);
387                    JsonObject properties = json.get("properties").getAsJsonObject();
388                    this.magnitude = properties.get("mag").getAsDouble();
389                    this.locationDescription = properties.get("place").getAsString();
390                    this.time = properties.get("time").getAsLong();
391                    this.url = properties.get("url").getAsString();
392                    if (properties.get("felt").isJsonNull()) {
393                            this.feltReports = -1;
394                    } else {
395                            this.feltReports = properties.get("felt").getAsInt();
396                    }
397                    if (properties.get("cdi").isJsonNull()) {
398                            this.maximumReportedIntensity = -1;
399                    } else {
400                            this.maximumReportedIntensity = properties.get("cdi").getAsDouble();
401                    }
402                    if (properties.get("mmi").isJsonNull()) {
403                            this.maximumEstimatedIntensity = -1;
404                    } else {
405                            this.maximumEstimatedIntensity = properties.get("mmi")
406                                            .getAsDouble();
407                    }
408                    if (properties.get("alert").isJsonNull()) {
409                            this.alertLevel = "";
410                    } else {
411                            this.alertLevel = properties.get("alert").getAsString();
412                    }
413                    this.eventSource = properties.get("status").getAsString();
414                    this.significance = properties.get("sig").getAsInt();
415                    this.id = json.get("id").getAsString();
416                    if (properties.get("dmin").isJsonNull()) {
417                            this.distanceToNearestStation = -1;
418                    } else {
419                            this.distanceToNearestStation = properties.get("dmin")
420                                            .getAsDouble();
421                    }
422                    if (properties.get("rms").isJsonNull()) {
423                            this.rootMeanSquare = -1;
424                    } else {
425                            this.rootMeanSquare = properties.get("rms").getAsDouble();
426                    }
427                    if (properties.get("gap").isJsonNull()) {
428                            this.gap = -1;
429                    } else {
430                            this.gap = properties.get("gap").getAsDouble();
431                    }
432            }

463             * @param alertLevel
464             *            A color string (one of "green", "yellow", "orange", "red")
465             *            indicating how dangerous the quake was, or null if the data is
466             *            not available. More information about this kind of alert is
467             *            available at http://earthquake.usgs.gov/research/pager/
468             * @param eventSource

476             *            maximum estimated intensity, felt reports, and estimated

480             * @param distance
481             *            Horizontal distance from the epicenter to the nearest station
482             *            (in degrees), or null if the data is not available. 1 degree
483             *            is approximately 111.2 kilometers. In general, the smaller
484             *            this number, the more reliable is the calculated depth of the
485             *            earthquake.
486             * @param rootMeanSquare
487             *            The root-mean-square (RMS) travel time residual, in sec, using
488             *            all weights. This parameter provides a measure of the fit of
489             *            the observed arrival times to the predicted arrival times for

508                    this.magnitude = magnitude;
