001    package realtimeweb.earthquakeservice.structured;
002    
003    import realtimeweb.earthquakeservice.main.AbstractEarthquakeService;
004    import realtimeweb.earthquakeservice.util.Util;
005    import java.util.HashMap;
006    import java.util.LinkedHashMap;
007    import com.google.gson.Gson;
008    import realtimeweb.earthquakeservice.json.JsonEarthquakeService;
009    import realtimeweb.earthquakeservice.json.JsonGetEarthquakesListener;
010    
011    /**
012     * Used to get data as built-in Java objects (HashMap, ArrayList, etc.).
013     */
014    public class StructuredEarthquakeService implements AbstractEarthquakeService {
015            private static StructuredEarthquakeService instance;
016            private JsonEarthquakeService jsonInstance;
017            private Gson gson;
018            /**
019             * **For internal use only!** Protected Constructor guards against instantiation.
020            
021             * @return 
022             */
023            protected  StructuredEarthquakeService() {
024                    this.jsonInstance = JsonEarthquakeService.getInstance();
025                    this.gson = new Gson();
026            }
027            
028            /**
029             * Retrieves the singleton instance.
030            
031             * @return StructuredEarthquakeService
032             */
033            public static StructuredEarthquakeService getInstance() {
034                    if (instance == null) {
035                            synchronized (StructuredEarthquakeService.class) {
036                                    if (instance == null) {
037                                            instance = new StructuredEarthquakeService();
038                                    }
039                            }
040                            
041                    }
042                    return instance;
043            }
044            
045            /**
046             * Establishes a connection to the online service. Requires an internet connection.
047            
048             */
049            @Override
050            public void connect() {
051                    jsonInstance.connect();
052            }
053            
054            /**
055             * Establishes that Business Search data should be retrieved locally. This does not require an internet connection.<br><br>If data is being retrieved locally, you must be sure that your parameters match locally stored data. Otherwise, you will get nothing in return.
056            
057             */
058            @Override
059            public void disconnect() {
060                    jsonInstance.disconnect();
061            }
062            
063            /**
064             * Retrieves information about earthquakes around the world.
065             * @param threshold A string indicating what kind of earthquakes to report. Must be either "significant" (only significant earthquakes), "all" (all earthquakes, regardless of significance), "4.5", "2.5", or "1.0". Note that for the last three, all earthquakes at and above that level will be reported.
066             * @param time A string indicating the time range of earthquakes to report. Must be either "hour" (only earthquakes in the past hour), "day" (only earthquakes that happened today), "week" (only earthquakes that happened in the past 7 days), or "month" (only earthquakes that happened in the past 30 days).
067             * @return HashMap<String, Object>
068             */
069            public HashMap<String, Object> getEarthquakes(String threshold, String time) throws Exception {
070                    return gson.fromJson(jsonInstance.getEarthquakes(threshold, time), LinkedHashMap.class);
071            }
072            
073            /**
074             * Retrieves information about earthquakes around the world.
075             * @param threshold A string indicating what kind of earthquakes to report. Must be either "significant" (only significant earthquakes), "all" (all earthquakes, regardless of significance), "4.5", "2.5", or "1.0". Note that for the last three, all earthquakes at and above that level will be reported.
076             * @param time A string indicating the time range of earthquakes to report. Must be either "hour" (only earthquakes in the past hour), "day" (only earthquakes that happened today), "week" (only earthquakes that happened in the past 7 days), or "month" (only earthquakes that happened in the past 30 days).
077             * @param callback The listener that will be given the data (or error)
078             */
079            public void getEarthquakes(String threshold, String time, final StructuredGetEarthquakesListener callback) {
080                    
081                    jsonInstance.getEarthquakes(threshold, time, new JsonGetEarthquakesListener() {
082                        @Override
083                        public void getEarthquakesFailed(Exception exception) {
084                            callback.getEarthquakesFailed(exception);
085                        }
086                        
087                        @Override
088                        public void getEarthquakesCompleted(String data) {
089                            callback.getEarthquakesCompleted(gson.fromJson(data, LinkedHashMap.class));
090                        }
091                    });
092                    
093            }
094            
095    }
